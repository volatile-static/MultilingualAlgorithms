52  static void compress(InputStream in, BitOutputStream out) throws IOException {
53  int[] initFreqs = new int[257];
54  Arrays.fill(initFreqs, 1);
55 
56  FrequencyTable freqs = new FrequencyTable(initFreqs);
57  HuffmanEncoder enc = new HuffmanEncoder(out);
58  enc.codeTree = freqs.buildCodeTree(); // Don't need to make canonical code because we don't transmit the code tree
59  int count = 0; // Number of bytes read from the input file
60  while (true) {
61  // Read and encode one byte
62  int symbol = in.read();
63  if (symbol == -1)
64  break;
65  enc.write(symbol);
66  count++;
67 
68  // Update the frequency table and possibly the code tree
69  freqs.increment(symbol);
70  if (count < 262144 && isPowerOf2(count) || count % 262144 == 0) // Update code tree
71  enc.codeTree = freqs.buildCodeTree();
72  if (count % 262144 == 0) // Reset frequency table
73  freqs = new FrequencyTable(initFreqs);
74  }
75  enc.write(256); // EOF
76  }
