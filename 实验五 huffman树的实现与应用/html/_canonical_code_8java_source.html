84  public CanonicalCode(int[] codeLens) {
85  // Check basic validity
86  Objects.requireNonNull(codeLens);
87  if (codeLens.length < 2)
88  throw new IllegalArgumentException("At least 2 symbols needed");
89  for (int cl : codeLens) {
90  if (cl < 0)
91  throw new IllegalArgumentException("Illegal code length");
92  }
93 
94  // Copy once and check for tree validity
95  codeLengths = codeLens.clone();
96  Arrays.sort(codeLengths);
97  int currentLevel = codeLengths[codeLengths.length - 1];
98  int numNodesAtLevel = 0;
99  for (int i = codeLengths.length - 1; i >= 0 && codeLengths[i] > 0; i--) {
100  int cl = codeLengths[i];
101  while (cl < currentLevel) {
102  if (numNodesAtLevel % 2 != 0)
103  throw new IllegalArgumentException("Under-full Huffman code tree");
104  numNodesAtLevel /= 2;
105  currentLevel--;
106  }
107  numNodesAtLevel++;
108  }
109  while (currentLevel > 0) {
110  if (numNodesAtLevel % 2 != 0)
111  throw new IllegalArgumentException("Under-full Huffman code tree");
112  numNodesAtLevel /= 2;
113  currentLevel--;
114  }
115  if (numNodesAtLevel < 1)
116  throw new IllegalArgumentException("Under-full Huffman code tree");
117  if (numNodesAtLevel > 1)
118  throw new IllegalArgumentException("Over-full Huffman code tree");
119 
120  // Copy again
121  System.arraycopy(codeLens, 0, codeLengths, 0, codeLens.length);
122  }

193  public CodeTree toCodeTree() {
194  List<Node> nodes = new ArrayList<Node>();
195  for (int i = max(codeLengths); i >= 0; i--) { // Descend through code lengths
196  if (nodes.size() % 2 != 0)
197  throw new AssertionError("Violation of canonical code invariants");
198  List<Node> newNodes = new ArrayList<Node>();
199 
200  // Add leaves for symbols with positive code length i
201  if (i > 0) {
202  for (int j = 0; j < codeLengths.length; j++) {
203  if (codeLengths[j] == i)
204  newNodes.add(new Leaf(j));
205  }
206  }
207 
208  // Merge pairs of nodes from the previous deeper layer
209  for (int j = 0; j < nodes.size(); j += 2)
210  newNodes.add(new InternalNode(nodes.get(j), nodes.get(j + 1)));
211  nodes = newNodes;
212  }
