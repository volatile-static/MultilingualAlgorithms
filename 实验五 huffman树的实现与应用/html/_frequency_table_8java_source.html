135  public CodeTree buildCodeTree() {
136  // Note that if two nodes have the same frequency, then the tie is broken
137  // by which tree contains the lowest symbol. Thus the algorithm has a
138  // deterministic output and does not rely on the queue to break ties.
139  Queue<NodeWithFrequency> pqueue = new PriorityQueue<NodeWithFrequency>();
140 
141  // Add leaves for symbols with non-zero frequency
142  for (int i = 0; i < frequencies.length; i++) {
143  if (frequencies[i] > 0)
144  pqueue.add(new NodeWithFrequency(new Leaf(i), i, frequencies[i]));
145  }
146 
147  // Pad with zero-frequency symbols until queue has at least 2 items
148  for (int i = 0; i < frequencies.length && pqueue.size() < 2; i++) {
149  if (frequencies[i] == 0)
150  pqueue.add(new NodeWithFrequency(new Leaf(i), i, 0));
151  }
152  if (pqueue.size() < 2)
153  throw new AssertionError();
154 
155  // Repeatedly tie together two nodes with the lowest frequency
156  while (pqueue.size() > 1) {
157  NodeWithFrequency x = pqueue.remove();
158  NodeWithFrequency y = pqueue.remove();
159  pqueue.add(new NodeWithFrequency(
160  new InternalNode(x.node, y.node),
161  Math.min(x.lowestSymbol, y.lowestSymbol),
162  x.frequency + y.frequency));
163  }
164 
165  // Return the remaining node
166  return new CodeTree((InternalNode)pqueue.remove().node, frequencies.length);
167  }
