39  // Read input file once to compute symbol frequencies.

41  FrequencyTable freqs = getFrequencies(inputFile);

59  // Returns a frequency table based on the bytes in the given file.
60  // Also contains an extra entry for symbol 256, whose frequency is set to 0.
61  private static FrequencyTable getFrequencies(File file) throws IOException {
62  FrequencyTable freqs = new FrequencyTable(new int[257]);
63  try (InputStream input = new BufferedInputStream(new FileInputStream(file))) {
64  while (true) {
65  int b = input.read();
66  if (b == -1)
67  break;
68  freqs.increment(b);
69  }
70  }
71  return freqs;
72  }

76  static void writeCodeLengthTable(BitOutputStream out, CanonicalCode canonCode) throws IOException {
77  for (int i = 0; i < canonCode.getSymbolLimit(); i++) {
78  int val = canonCode.getCodeLength(i);
79  // For this file format, we only support codes up to 255 bits long
80  if (val >= 256)
81  throw new RuntimeException("The code for a symbol is too long");
82 
83  // Write value as 8 bits in big endian
84  for (int j = 7; j >= 0; j--)
85  out.write((val >>> j) & 1);
86  }
