51  FrequencyTable freqs = new FrequencyTable(initFreqs);

53  dec.codeTree = freqs.buildCodeTree(); // Use same algorithm as the compressor
54  int count = 0; // Number of bytes written to the output file
55  while (true) {
56  // Decode and write one byte
57  int symbol = dec.read();
58  if (symbol == 256) // EOF symbol
59  break;
60  out.write(symbol);
61  count++;

64  freqs.increment(symbol);
65  if (count < 262144 && isPowerOf2(count) || count % 262144 == 0) // Update code tree
66  dec.codeTree = freqs.buildCodeTree();
67  if (count % 262144 == 0) // Reset frequency table
68  freqs = new FrequencyTable(initFreqs);
